3.Abstract Class:

package shapes;
abstract class Shapes {
    int length=2;
    int breadth=8;
    int height=6;
    double circle,rectangle;
    int radius=4;
    double pi=3.14;
    abstract double calculate();
}
public class circle extends Shapes{
    double calculate()
    {
        return(pi*radius*radius);
    }
}
public class rectangle extends Shapes{
    double calculate()
    {
        return(length*breadth*height);
    }
}
public class display {
    public static void main(String args[]){
    Shapes s1=new circle();
    System.out.println("Area of circle:"+s1.calculate());
    Shapes s2=new rectangle();
    System.out.println("Area of rectangle:"+s2.calculate());
}
}

Output:

Area of circle:50.24
Area of rectangle96.0

4.Method Overloading:

package bank;
public class Bank {
   String name,address;
   int accno,balance;
   public void display(){
       System.out.println("Name:"+name);
       System.out.println("Accno:"+accno);
       System.out.println("Address:"+address);
   }
}
public class creditcard extends Bank{
    public void display(){
        System.out.println("Details of creditcard:");
        balance+=10000;
        System.out.println("Balance:"+balance);

    }   
}
public class debitcard extends Bank{
     public void display(){
        System.out.println("Details of debitcard:");
        balance-=10000;
        System.out.println("Balance:"+balance);
     }
}
public class netbanking extends Bank{
     public void display(){
        System.out.println("Details of netbanking:");
        System.out.println("Balance:"+balance);
     }
}
import java.util.Scanner;
public class main {
    public static void main(String args[]){
    Scanner sc=new Scanner(System.in);
    Bank b=new Bank();
        System.out.println("Enter name:");

        System.out.println("Enter the address:");
        b.address=sc.nextLine();

        creditcard c=new creditcard();
        c.name=b.name;

5.Using Multilevel inheitance to fine Netpay & Grosspay:

package one;
import java.util.Scanner;
public class One {
    int ID,salary;
    double PF,HRA;
    double incomeTax;
    double professionalTax;
    String name;
    double netpay,grosspay;
    void insert()
    {
       Scanner sc=new Scanner(System.in);
       System.out.println("Enter the name:");
       name=sc.nextLine();
       System.out.println("Enter the id:");
       ID=sc.nextInt();
       System.out.println("Enter the salary:");
       salary=sc.nextInt();
    }
}
package one;
public class two extends One{
    void display()
    {
       System.out.println("Name="+name);
       System.out.println("ID="+ID);
       System.out.println("Salary="+salary);
    }
    void calculate()
    {
        PF=(salary*0.10);
        HRA=(salary*0.05);
        incomeTax=(salary*0.12);
        professionalTax=(salary*0.10);
        grosspay=salary+HRA;
        netpay=grosspay-(incomeTax+professionalTax+PF); 
    }
}
package one;
public class three extends two {
    void print()
    {
        System.out.println("Employee salary:");
        System.out.println("Grosspay="+grosspay);
        System.out.println("Netpay="+netpay);
    }
}
package one;
public class main {
    public static void main(String args[])
    {
       three A=new three();
       A.insert();
       A.display();
       A.calculate();
       A.print();
    }
}
