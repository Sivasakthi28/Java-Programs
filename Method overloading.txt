package bank;
public class Bank {
   String name,address;
   int accno,balance;
   public void display(){
       System.out.println("Name:"+name);
       System.out.println("Accno:"+accno);
       System.out.println("Address:"+address);
   }
}
public class creditcard extends Bank{
    public void display(){
        System.out.println("Details of creditcard:");
        balance+=10000;
        System.out.println("Balance:"+balance);

    }   
}
public class debitcard extends Bank{
     public void display(){
        System.out.println("Details of debitcard:");
        balance-=10000;
        System.out.println("Balance:"+balance);
     }
}
public class netbanking extends Bank{
     public void display(){
        System.out.println("Details of netbanking:");
        System.out.println("Balance in:"+balance);
     }
}
import java.util.Scanner;
public class main {
    public static void main(String args[]){
    Scanner sc=new Scanner(System.in);
    Bank b=new Bank();
        System.out.println("Enter name:");
        b.name=sc.nextLine();
        System.out.println("Enter acc_no:");
        b.accno=sc.nextInt();
        System.out.println("Enter the address:");
        b.address=sc.nextLine();
        System.out.println("Enter the balance:");
        b.balance=sc.nextInt();
        creditcard c=new creditcard();
        c.name=b.name;
        c.accno=b.accno;
        c.address=b.address;
        c.balance=b.balance;
        b.display();
        debitcard d=new debitcard();
        d.name=b.name;
        d.accno=b.accno;
        d.address=b.address;
        d.balance=b.balance;
        d.display();
        netbanking n=new netbanking();
        n.name=b.name;
        n.accno=b.accno;
        n.address=b.address;
        n.balance=b.balance;
        n.display();
}
}
